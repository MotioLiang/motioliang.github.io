SelectDateRange 组件用于进行选择日期时间段预约

1、组件目前实现的功能点如下

  • 实现 任意选择日期范围、取消选择、确认选择、
  • 实现 默认占用

2、在 components 文件中创建 SelectDateRange.vue 文件

<script setup>
import { reactive, watch } from 'vue'

const emit = defineEmits(['handle-cancel', 'handle-comfirm'])

const props = defineProps({
    // 默认当前时间，可选择
    dateList: {
        type: Array,
        default: () => []
    },

    // 预约的时间段，准备预约的时间段
    defaultList: {
        type: Array,
        default: () => []
    }
})

const state = reactive({
    dateList: [],
    list: [],
    activeData: { rowId: '', date: '', dateIndex: 0 },
    isShowTip: false,
    popover: {
        isMiddle: false,
        isShow: false,
        id: ''
    }
})

watch(
    () => props.dateList,
    newVal => {
        state.dateList = JSON.parse(JSON.stringify(newVal))
    },
    { deep: true, immediate: true }
)

watch(
    () => props.defaultList,
    newVal => {
        const list = JSON.parse(JSON.stringify(newVal))
        list.forEach(item => {
            item.date = []
            item.middleDate = []
        })
        state.list = list
    },
    { deep: true, immediate: true }
)

function handleCellAction(row, index, date, i) {
    if (!state.activeData.rowId) {
        state.activeData.rowId = row.id
        state.activeData.date = date
        state.activeData.dateIndex = i
        state.isShowTip = true

        if (!state.list[index].date.find(v => v.value === date)) {
            state.list[index].date.push({ value: date, index: i })
            const data = getCurrentDateRange(row, date, i)
            state.activeData.beginIndex = data.beginIndex
            state.activeData.endIndex = data.endIndex
        }
    } else {
        if (state.activeData.rowId === row.id) {
            const parasm = {
                id: +new Date(),
                list: state.list[index].date
            }
            state.list[index].middleDate.push(parasm)
            state.list[index].date = []

            state.activeData.rowId = ''
            state.activeData.date = ''
            state.activeData.dateIndex = ''

            state.popover.isMiddle = true
            state.popover.id = parasm.id

            state.activeData.beginIndex = 0
            state.activeData.endIndex = 0
        }
    }
}

function getCurrentDateRange(row, date, index) {
    const data = {}

    for (let i = index; i < state.dateList.length; i++) {
        const a = row.disabledDate.find(v => v.list.includes(state.dateList[i]))
        const b = row.chooseDate.find(v => v.list.find(k => k.value === state.dateList[i]))
        if (a || b) {
            data.endIndex = i
            break
        }
    }

    if (data.endIndex === undefined) data.endIndex = state.dateList.length - 1

    for (let i = index; i >= 0; i--) {
        const a = row.disabledDate.find(v => v.list.includes(state.dateList[i]))
        const b = row.chooseDate.find(v => v.list.find(k => k.value === state.dateList[i]))
        if (a || b) {
            data.beginIndex = i - 1
            break
        }
    }

    if (data.beginIndex === undefined) data.beginIndex = 0

    return data
}

function onTdMouseenter(row, index, date, i) {
    if (row.id !== state.activeData.rowId) return
    if (state.activeData.beginIndex > i) return
    if (state.activeData.endIndex < i) return

    if (date === state.activeData.date) {
        state.list[index].date = [{ value: state.activeData.date, index: i }]
        return
    }

    const list = []
    if (state.activeData.dateIndex < i) {
        state.dateList.forEach((item, index) => {
            if (state.activeData.dateIndex <= index && index <= i) {
                list.push({ value: item, index })
            }
        })
    } else if (state.activeData.dateIndex > i) {
        state.dateList.forEach((item, index) => {
            if (state.activeData.dateIndex >= index && index >= i) {
                list.push({ value: item, index })
            }
        })
    }
    state.list[index].date = list
}

function getDisabledTdMask(data) {
    const beginIndex = state.dateList.indexOf(data.list[0])
    return {
        width: data.list.length * 70 - 3 + 'px',
        left: beginIndex * 70 + 1 + 'px'
    }
}

function getChooseTdMask(data) {
    const beginIndex = data.list[0].index
    return {
        width: data.list.length * 70 - 3 + 'px',
        left: beginIndex * 70 + 1 + 'px'
    }
}

function getChooseDateRange(data) {
    return `${data.list[0].value} ~ ${data.list[data.list.length - 1].value}`
}

function openPopover(item) {
    if (state.popover.isShow) return
    if (state.popover.isMiddle) return
    state.popover.id = item.id
    state.popover.isShow = true
}

function closeReservePopover() {
    state.popover.isShow = false
    state.popover.isMiddle = false
    state.popover.id = ''
}

function removeReservePopover(row) {
    closeReservePopover()
    row.middleDate = []
}

function cancelReserve(row, index, item) {
    closeReservePopover()
    row.chooseDate = row.chooseDate.filter(v => v.id !== item.id)
    emit('handle-cancel', formatData(state.list))
}

function comfirmReserve(row) {
    row.chooseDate.push(row.middleDate[row.middleDate.length - 1])
    row.middleDate = []
    closeReservePopover()
    emit('handle-comfirm', formatData(state.list))
}

function formatData(data) {
    const list = JSON.parse(JSON.stringify(data))
    list.forEach(item => {
        delete item.date
        delete item.middleDate
        item.chooseDate = item.chooseDate.map(v => v.list.map(k => k.valuess))
    })
    return list
}
</script>
